#include<bits/stdc++.h>
using namespace std;
int main(){
    int n;
    cin >> n;
    vector<vector<int>> graph(n , vector<int>(n));
    for(int i = 0  ; i < n ; i++){
        for(int j = 0 ; j < n ; j++){
            cin >> graph[i][j];
        }
    }
    for(int k = 0 ; k < n ; k++){
        for(int i = 0 ; i < n ; i++){
            for(int j = 0 ; j < n ; j++){
                if(graph[i][k] != INT_MAX && graph[k][j] != INT_MAX &&graph[i][k] + graph[k][j] < graph[i][j]){
                    graph[i][j] = graph[i][k] + graph[k][j];
                }
            }
        }
    }
    for(int i = 0 ; i < n ; i++){
        for(int j = 0 ; j < n ; j++){
           if (graph[i][j] == INT_MAX){    // change int_max according to query to print inf 
                cout << "INF ";
            }else{
                cout << graph[i][j] << " ";
            }
        }
        cout<<endl;
    }
    return 0 ;
}
