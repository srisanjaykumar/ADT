#include <bits/stdc++.h>
using namespace std;

int n, fixedRow, fixedCol;
int pos[100];
bool found = false;

bool safe(int r, int c) {
    for (int i = 0; i < r; i++)
        if (pos[i] == c || pos[i] + i == c + r || pos[i] - i == c - r)
            return false;
    return true;
}void solve(int r) {
    if (r == n) {
        found = true;
        for (int i = 0; i < n; i++)   {
            for (int j = 0; j < n; j++) {
                if (pos[i] == j)
                    cout << "Q ";
                else
                    cout << ". ";
            }
            cout << "\n";
        }
        return;
    }

    if (r == fixedRow) {
        if (safe(r, fixedCol)) {
            pos[r] = fixedCol;
            solve(r + 1);
        }
        return;
    }

    for (int c = 0; c < n; c++) {
        if (safe(r, c)) {
            pos[r] = c;
            solve(r + 1);
        }
    }
}

int main() {
    cin >> n >> fixedRow >> fixedCol;
    solve(0);
    if (!found) cout << "No solution\n";
}
