#include<bits/stdc++.h>
using namespace std;
int main(){
    int N , Q;
    cin >> N >> Q;
    vector<int> arr(N);
    for(int i  = 0 ; i < N  ; i++){
        cin >> arr[i];
    }
    for(int q = 0 ; q < Q ; q++){
        char type;
        cin >> type;
        if(type  == 'S'){
            int L , R;
            cin >> L >> R;
            int sum = 0 ;
            for(int i = L ; i <= R ; i++){
                sum += arr[i];
            }
            cout<<sum<<endl;
        }
        if(type == 'U'){
            int index , value;
            cin >> index >> value;
            arr[index] = value;
        }
    }
    return 0;
}
