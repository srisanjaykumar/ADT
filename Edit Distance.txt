#include<bits/stdc++.h>
using namespace std;
int main(){
    string a , b ;
    int  ins , upt ,del ;
    cin>>a>>b>>ins>>upt>>del;
    int n = a.length();
    int m = b.length();
    vector<vector<int>> dp(n + 1 , vector<int>(m + 1 , 0));
    for(int i = 0 ; i <= n ; i++){
        dp[i][0] = i * del;
        for(int j = 0 ; j <= m ; j++){
            dp[0][j] = j * ins;
        }
    }
    for(int i = 1 ; i <= n ; i++){
        for(int j = 1 ; j <= m ; j++){
            if(a[i - 1] == b[j - 1]){
                dp[i][j] = dp[i - 1][j - 1];
            }
            else{
                int insert = dp[i][j - 1] + ins;
                int d = dp[i - 1][j] + del;
                int update = dp[i - 1][j - 1 ] + upt;
                dp[i][j]=min({insert , d , update});
            }
        }
    }
    cout<<dp[n][m]<<endl;
    return 0;
}
